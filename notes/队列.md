# 队列(Queue)

## 队列的基本概念
只许在一端进行插入，在另一端进行删除的线性表(FIFO)

**基本操作**：创、销、增、删、改、查、判空    

**队头指针**：指向头元素  
**队尾指针**：指向下一个应该插入的位置  

## 队列的顺序实现  
```
#define MaxSize 10

//结构体
typedef struct
{
    ElemType data[MaxSize];
    int front, rear;
}SqQueue;

//初始化
bool InitQueue(SqQueue &Q)
{
    if(NULL == Q)
    {
        return false;
    }
    
    Q.rear  = 0;
    Q.front = 0;
    
    return true;
}

//判空
bool isQueueEmpty(SqQueue Q)
{
    return (Q.rear == Q.front);
}

//判满
**三种办法**  
1).牺牲一个存储空间  (判断元素个数：(rear+MaxSize-front)%MaxSize
2).结构体中增加一个表示元素个数的数据成员
3).结构体中增加一个tag数据成员用于标识最近一次操作是入队还是出队
这里使用方法1
bool isQueueFilled(SqQueue Q)
{
    return ((Q.rear+1)%MaxSize == Q.front);
}

//入队
bool EnQueue(SqQueue &Q, ElemType e)
{
    if(NULL == Q)
    {
        return false;
    }
    
    if(isQueueFilled(Q))
    {
        return false;
    }
    
    Q.data[Q.rear] = e;
    Q.rear = (Q.rear + 1) % MaxSize;
    
    return true;
}

//出队
bool DeQueue(SqQueue &Q, ElemType &e)
{
    if(NULL == Q)
    {
        return false;
    }
    
    if(isQueueEmpty(Q))
    {
        return false;
    }
    
    e = Q.data[Q.front];
    //清空Q.data[Q.front - 1]
    Q.front = (Q.front + 1) % MaxSize;
    
    return true;
}

//查找
只查队头
bool GetHeadQ(SqQueue Q, ElemType &e)
{
    if(isQueueEmpty(Q))
    {
        return false;
    }
    
    e = Q.data[Q.front];
    
    return true;
}
```

## 队列的链式实现

### 带头结点
```
//结点结构体
typedef struct LNode
{
    ElemType data;
    struct LNode *next;
}LNode;

//链式队列
typedef struct
{
    LNode *rear, *front;
}LinkQueue;

//初始化
bool InitLQueue(LinkQueue &Q)
{
    //增加头结点
    Q.rear  = (LNode*)malloc(sizeof(LNode));
    Q.front = Q.rear;
    Q.rear->next = NULL;

    return true;
}

//判空
//case1:
bool IsEmpty(LinkQueue Q)
{
    return (Q.front == Q.rear);
}
//case2:
bool IsEmpty(LinkQueue Q)
{
    return (NULL == Q.front->next);
}

//入队
bool EnLQueue(LinkQueue &Q, ElemType e)
{
    LNode *p = (LNode*)malloc(sizeof(LNode));
    if(NULL == p)
    {
        return false;
    }
    
    p->data = e;
    p->next = NULL;
    Q.rear->next = p;
    Q.rear = p;
    
    return true;
}

//出队
bool DeQueue(LinkQueue &Q, ElemType &e)
{
    if(IsEmpty(Q))
    {
        return false;
    }
    
    LNode *p = &Q.front->next;
    e = p->data;
    Q.front->next = Q.front->next->next;
    if(Q.rear == p)
    {
        Q.rear = Q.front;
    }
    free(p);
    p = NULL;
    return true;
}
```

### 不带头结点
```
//结构体
typedef struct LNode 
{
    ElemType data;
    struct LNode *next;
}LNode;

typedef struct
{
    LNode *rear, *front;
}LinkQueue;

//判空
bool IsEmpty(LinkQueue Q)
{
    return (Q.front == NULL);
}

//初始化
bool InitQueue(LinkQueue &Q)
{
    Q.rear = Q.front = NULL;
    
    return true;
}

//入队
bool EnQueue(LinkQueue &Q, ElemType e)
{
    LNode *p = (LNode*)malloc(sizeof(LNode));
    if(NULL == p)
    {
        return false;
    }
    
    p->data = e;
    p->next = NULL;
    if(IsEmpty(Q))
    {
        Q.front = p;
    }
    else
    {
        Q.rear->next = p;
    }
    Q.rear = p;
    
    return true;
}

//出队
bool DeQueue(LinkQueue &Q, ElemType &e)
{
    if(IsEmpty(Q))
    {
        return false;
    }
    
    LNode *p = Q.front;
    e = p->data;
    if(NULL == p->next)
    {
        Q.rear = Q.front = NULL;
    }
    else
    {
        Q.front = p->next;
    }
    free(p);
    
    return true;
}
```

## 双端队列

**只允许从两端插入和删除的线性表**

**变种**  
输入受限的双端队列：只允许一端输入  
输出受限的双端队列：只允许一端输出  

**输出序列合法性**  
**1)输入受限**  
根据第一个出队的元素可以知道在该元素之前的元素的入队顺序，即知道了元素在队列中的位置，试着从两边出队看能不能得到输出序列  

**2)输出受限**  
根据第一个出队的元素可以知道在该元素之前的元素的出队顺序进一步可以知道元素在队列中的位置，试着从两边入队看能不能得到输入序列
